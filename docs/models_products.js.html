   * Post one record

   * @returns Resolved promise with the record

    let newRecord = new schema(entry);
    return newRecord.save();
  }

   * Update one record
   * @param {string} _id
   * @param {object} record

    const options = {new: true};

  }

   * @param {string} _id
   * @param {object} record
   * @returns Empty object
   * @memberof Product
   */
  delete(_id) {
    const query = {_id:_id}
    const options = {new: true};
    return schema.findByIdAndDelete(query);
  }

}
